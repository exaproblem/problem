using Leonteva.databases;

public MainWindow()
        {
            InitializeComponent();
            filterComboBox.ItemsSource = Instance.GetContext().ProductType.ToList().Prepend(new ProductType() { Title = "выберите тип" });
            filterComboBox.SelectedIndex = 0;

            sortComboBox.Items.Add("нет сортировки");
            sortComboBox.Items.Add("по убыванию");
            sortComboBox.Items.Add("по возрастанию");
            sortComboBox.SelectedIndex = 0;

            updateData();
        }

        private void updateData()
        {
            var listBoxList = Instance.GetContext().Product.ToList();

            // поиск
            if (!string.IsNullOrEmpty(poisk.Text))
                listBoxList = listBoxList.Where(p => p.Title.Trim().Contains(poisk.Text)).ToList();

            // фильтрация
            if (filterComboBox.SelectedIndex > 0)
                listBoxList = listBoxList.Where((p) => p.ProductTypeID.Equals(filterComboBox.SelectedIndex)).ToList();

            // sort
            switch (sortComboBox.SelectedIndex)
            {
                case 0:
                    break;
                case 1:
                    listBoxList = listBoxList.OrderBy(p => p.Title).ToList();
                    break;
                case 2:
                    listBoxList = listBoxList.OrderByDescending(p => p.Title).ToList();
                    break;
            }

            // присвоим модифицированные временные списки нашим элементам вывода
            listbox.ItemsSource = listBoxList;
        }

        private void sortComboBox_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            updateData();
        }

        private void filterComboBox_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            updateData();
        }

        private void poisk_TextChanged(object sender, TextChangedEventArgs e)
        {
            updateData();
        }

7)создать в папке common Converter.cs

using demoExam.databases;
using System;
using System.Globalization;
using System.Linq;
using System.Windows.Data;
using System.Windows.Media.Imaging;

namespace demoExam.common
{
    [ValueConversion(typeof(Product), typeof(string))]
    public class ConventerImage : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            Product product = value as Product;
            try 
            {
                if (product == null || product.Image == string.Empty)
                    return new BitmapImage(new Uri(System.IO.Path.GetFullPath("../../resources/picture.png")));
                return new BitmapImage(new Uri(System.IO.Path.GetFullPath($"../../resources/{product.Image}")));
            }
            catch
            {
                return new BitmapImage(new Uri(System.IO.Path.GetFullPath("../../resources/picture.png")));
            }
            
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value;
        }
    }

    [ValueConversion(typeof(Product), typeof(string))]
    public class ConventerType : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            Product product = value as Product;
            if (product == null)
                return "none";
            return $"{product.ProductType.Title} | {product.Title}";
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value;
        }
    }

    [ValueConversion(typeof(Product), typeof(string))]
    public class ConventerArticle : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            Product product = value as Product;
            if (product == null)
                return "none";
            return $"арт. {product.ArticleNumber}";
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value;
        }
    }

    [ValueConversion(typeof(Product), typeof(string))]
    public class ConventerMaterial : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            Product product = value as Product;
            if (product == null)
                return "none";
            string output = string.Empty;
            foreach (string material in product.ProductMaterial.Select(p => p.Material.Title).ToList())
                output += $"{material}; ";
            return $"{output}";
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value;
        }
    }

    [ValueConversion(typeof(Product), typeof(string))]
    public class ConventerCost : IValueConverter
    {
        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            Product product = value as Product;
            if (product == null)
                return "none";
            return $"{product.MinCostForAgent}";
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value;
        }
    }
}
